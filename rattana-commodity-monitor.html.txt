function parseCsv(text){
  const parse = r => { const c=[]; let cur='',q=false; for(const ch of r+','){ if(ch==='"'){q=!q}else if(ch===','&&!q){c.push(cur.trim());cur=''}else{cur+=ch} } return c; };
  const rows=text.trim().split('\n'), h=parse(rows[0]);
  return rows.slice(1).map(r=>Object.fromEntries(h.map((k,i)=>[k,(parse(r)[i]||'')])));
}
async function loadFromSheets(sheetId, sheetName){
  const url = `https://docs.google.com/spreadsheets/d/${sheetId}/gviz/tq?tqx=out:csv&sheet=${encodeURIComponent(sheetName)}`;
  const res = await fetch(url);
  if(!res.ok) throw new Error('fetch failed');
  return parseCsv(await res.text());
}

async function handleGoogleLogin(r){
  showSpinner('Verifying...');
  try {
    const email = JSON.parse(atob(r.credential.split('.')[1])).email.toLowerCase().trim();
    const users = await loadFromSheets(AUTH_CONFIG.usersSheetId, AUTH_CONFIG.usersSheetName);
    const user = users.find(u => (u['E-mail']||'').toLowerCase().trim() === email);
    if(!user){ showLoginError('Email not found in Rattana user list. Contact Admin.'); return; }
    if((user['Status']||'').trim() !== 'Active'){ showLoginError('Account suspended. Contact Admin.'); return; }
    currentUser = {
      email, name:user['ชื่อ - สกุล']||email, nickname:user['ชื่อเล่น']||'',
      role:parseInt(user['User Role'])||1, dept:user['แผนก']||'',
      company:user['บริษัท']||'', warehouse:user['W']||'', empId:user['รหัสพนักงาน']||'',
    };
    sessionStorage.setItem('rattanaUser', JSON.stringify(currentUser));
    enterApp();
  } catch(e) {
    console.error(e);
    showLoginError('Login error. Check network & try again.');
  } finally { hideSpinner(); }
}

}

async function refreshData(){

function switchTab(ev, key){
  document.querySelectorAll('.tab-bar button').forEach(b=>b.classList.remove('active'));
  ev.target.classList.add('active');
  const tableWrap=document.getElementById('tableWrap');
  if(key==='table'){ document.getElementById('gridAll').style.display='none'; tableWrap.style.display='block'; return; }
  tableWrap.style.display = key==='all' ? 'block' : 'none';
  document.getElementById('gridAll').style.display='grid';
  document.querySelectorAll('#gridAll .card').forEach(c=>{
    c.style.display = (key==='all' || (c.dataset.tab||'').includes(key)) ? '' : 'none';
  });
}
